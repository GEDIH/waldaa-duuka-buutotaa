## 4. 📅 **Event Calendar**

### **Problem**: Events are static text
### **Solution**: Interactive calendar

```html
<!-- Add to your events section -->
<div class="calendar-container">
  <div class="calendar-header">
    <button id="prevMonth">‹</button>
    <h3 id="currentMonth"></h3>
    <button id="nextMonth">›</button>
  </div>
  <div class="calendar-grid" id="calendarGrid">
    <!-- Calendar will be generated by JavaScript -->
  </div>
  <div class="event-details" id="eventDetails">
    <h4>Taateewwan Filataman</h4>
    <div id="selectedEvents"></div>
  </div>
</div>

<style>
.calendar-container {
  background: white;
  border-radius: 15px;
  padding: 2rem;
  margin: 2rem 0;
  box-shadow: 0 5px 15px rgba(0,0,0,0.1);
}

.calendar-header {
  display: flex;
  justify-content: space-between;
  align-items: center;
  margin-bottom: 1rem;
}

.calendar-grid {
  display: grid;
  grid-template-columns: repeat(7, 1fr);
  gap: 1px;
  background: #e9ecef;
  border-radius: 10px;
  overflow: hidden;
}

.calendar-day {
  background: white;
  padding: 1rem;
  text-align: center;
  cursor: pointer;
  transition: background 0.3s ease;
}

.calendar-day:hover {
  background: #f8f9fa;
}

.calendar-day.has-event {
  background: #d4af37;
  color: white;
}

.event-details {
  margin-top: 2rem;
  padding: 1rem;
  background: #f8f9fa;
  border-radius: 10px;
}
</style>

<script>
class EventCalendar {
  constructor() {
    this.currentDate = new Date();
    this.events = {
      '2024-12-01': ['Yaa\'ii Ji\'aa - Sa\'a 2:00'],
      '2024-12-07': ['Kadhannaa Walii - Sa\'a 12:00'],
      '2024-12-14': ['Kadhannaa Walii - Sa\'a 12:00'],
      '2024-12-21': ['Kadhannaa Walii - Sa\'a 12:00'],
      '2024-12-25': ['Ayyaana Dhalootaa Kiristoos'],
      '2024-12-28': ['Kadhannaa Walii - Sa\'a 12:00']
    };
    this.init();
  }
  
  init() {
    this.renderCalendar();
    this.bindEvents();
  }
  
  renderCalendar() {
    const year = this.currentDate.getFullYear();
    const month = this.currentDate.getMonth();
    
    // Update month display
    const monthNames = [
      'Amajjii', 'Guraandhala', 'Bitooteessa', 'Elba', 'Caamsa', 'Waxabajjii',
      'Adooleessa', 'Hagayya', 'Fuulbana', 'Onkololeessa', 'Sadaasa', 'Muddee'
    ];
    document.getElementById('currentMonth').textContent = `${monthNames[month]} ${year}`;
    
    // Generate calendar days
    const firstDay = new Date(year, month, 1).getDay();
    const daysInMonth = new Date(year, month + 1, 0).getDate();
    
    let calendarHTML = '';
    
    // Day headers
    const dayHeaders = ['Dil', 'Wix', 'Kib', 'Rob', 'Kam', 'Jim', 'San'];
    dayHeaders.forEach(day => {
      calendarHTML += `<div class="calendar-day-header">${day}</div>`;
    });
    
    // Empty cells for days before month starts
    for (let i = 0; i < firstDay; i++) {
      calendarHTML += '<div class="calendar-day empty"></div>';
    }
    
    // Days of the month
    for (let day = 1; day <= daysInMonth; day++) {
      const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
      const hasEvent = this.events[dateStr] ? 'has-event' : '';
      calendarHTML += `<div class="calendar-day ${hasEvent}" data-date="${dateStr}">${day}</div>`;
    }
    
    document.getElementById('calendarGrid').innerHTML = calendarHTML;
  }
  
  bindEvents() {
    document.getElementById('prevMonth').addEventListener('click', () => {
      this.currentDate.setMonth(this.currentDate.getMonth() - 1);
      this.renderCalendar();
    });
    
    document.getElementById('nextMonth').addEventListener('click', () => {
      this.currentDate.setMonth(this.currentDate.getMonth() + 1);
      this.renderCalendar();
    });
    
    document.getElementById('calendarGrid').addEventListener('click', (e) => {
      if (e.target.classList.contains('calendar-day') && e.target.dataset.date) {
        this.showEvents(e.target.dataset.date);
      }
    });
  }
  
  showEvents(date) {
    const events = this.events[date] || ['Taateewwan hin jiran'];
    const eventsHTML = events.map(event => `<p>• ${event}</p>`).join('');
    document.getElementById('selectedEvents').innerHTML = eventsHTML;
  }
}

// Initialize calendar
document.addEventListener('DOMContentLoaded', () => {
  new EventCalendar();
});
</script>
```

## 5. 🔔 **Push Notifications**

### **Problem**: No way to notify users of updates
### **Solution**: Web push notifications

```javascript
// Add this to your main JavaScript
class NotificationManager {
  constructor() {
    this.init();
  }
  
  async init() {
    if ('serviceWorker' in navigator && 'PushManager' in window) {
      await this.registerServiceWorker();
      this.setupNotificationButton();
    }
  }
  
  async registerServiceWorker() {
    try {
      const registration = await navigator.serviceWorker.register('/sw.js');
      console.log('Service Worker registered');
    } catch (error) {
      console.log('Service Worker registration failed');
    }
  }
  
  setupNotificationButton() {
    // Add notification permission button to your site
    const notifyBtn = document.createElement('button');
    notifyBtn.textContent = 'Beeksisa Fudhachuu Barbaadduu?';
    notifyBtn.className = 'btn btn-secondary';
    notifyBtn.style.margin = '1rem';
    
    notifyBtn.addEventListener('click', () => {
      this.requestNotificationPermission();
    });
    
    // Add to your hero section or footer
    document.querySelector('.hero-buttons').appendChild(notifyBtn);
  }
  
  async requestNotificationPermission() {
    const permission = await Notification.requestPermission();
    
    if (permission === 'granted') {
      this.showNotification('Galatoomaa!', 'Beeksisa fudhachuuf galmeeffamtan.');
    }
  }
  
  showNotification(title, body) {
    if (Notification.permission === 'granted') {
      new Notification(title, {
        body: body,
        icon: '/images/WDB_LOGO_CIRCLE.png',
        badge: '/images/WDB_LOGO_CIRCLE.png'
      });
    }
  }
}

// Initialize notifications
new NotificationManager();
```

## 6. 📊 **Analytics Integration**

### **Problem**: No data on website usage
### **Solution**: Add Google Analytics

```html
<!-- Add before closing </head> tag -->
<script async src="https://www.googletagmanager.com/gtag/js?id=GA_MEASUREMENT_ID"></script>
<script>
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}
  gtag('js', new Date());
  gtag('config', 'GA_MEASUREMENT_ID');
  
  // Track custom events
  function trackEvent(action, category, label) {
    gtag('event', action, {
      event_category: category,
      event_label: label
    });
  }
  
  // Track chat interactions
  document.addEventListener('DOMContentLoaded', function() {
    // Track when chat is opened
    document.getElementById('chatIcon').addEventListener('click', function() {
      trackEvent('chat_opened', 'engagement', 'chat_widget');
    });
    
    // Track gallery views
    document.querySelectorAll('.gallery-item img').forEach(img => {
      img.addEventListener('click', function() {
        trackEvent('image_viewed', 'gallery', this.alt);
      });
    });
  });
</script>
```
