function fmtDate(iso) {
  if (!iso) return '—';
  return new Date(iso).toLocaleDateString('om-ET', {year:'numeric',month:'short',day:'numeric'});
}
function fmtDateTime(iso) {
  if (!iso) return '—';
  return new Date(iso).toLocaleString('om-ET', {year:'numeric',month:'short',day:'numeric',hour:'2-digit',minute:'2-digit'});
}

function doLogin() {
  if (loginAttempts >= 5) {
    showLoginError('Seensa cufame. Daqiiqaa 1 booda yaali.');
    return;
  }
  const u = document.getElementById('loginUser').value.trim();
  const p = document.getElementById('loginPass').value;
  const creds = getCreds();
  if (u === creds.user && p === creds.pass) {
    loginAttempts = 0;
    setData(SESSION_KEY, { loggedIn: true, time: Date.now() });
    document.getElementById('loginScreen').style.display = 'none';
    document.getElementById('adminApp').style.display = 'block';
    startSession();
    loadAll();
  } else {
    loginAttempts++;
    const left = 5 - loginAttempts;
    showLoginError('Fayyadamaa ykn jecha darbii dogoggoraa dha.');
    document.getElementById('loginAttempts').textContent =
      left > 0 ? `Yaaliiwwan ${left} hafan` : 'Seensa cufame!';
    if (loginAttempts >= 5)
      setTimeout(() => { loginAttempts = 0; document.getElementById('loginAttempts').textContent = ''; }, 60000);
  }
}

  const el = document.getElementById('loginError');

function doLogout() {
  // Show confirmation modal instead of logging out directly
  const modal = document.getElementById('logoutModal');
  modal.style.display = 'flex';
  // Close on backdrop click
  modal.onclick = function(e) { if (e.target === modal) cancelLogout(); };
  // Close on Escape key
  document._logoutEsc = function(e) { if (e.key === 'Escape') cancelLogout(); };
  document.addEventListener('keydown', document._logoutEsc);
}

function cancelLogout() {
  document.getElementById('logoutModal').style.display = 'none';
  document.removeEventListener('keydown', document._logoutEsc);
}

let pendingFiles = [];

function handleFiles(files) {
  pendingFiles = Array.from(files).filter(f => f.type.startsWith('image/'));
  const grid = document.getElementById('previewGrid');
  const actions = document.getElementById('uploadActions');
  grid.innerHTML = '';
  if (!pendingFiles.length) return;
  pendingFiles.forEach((file, i) => {
    const reader = new FileReader();
    reader.onload = e => {
      const card = document.createElement('div');
      card.className = 'preview-card';
      card.innerHTML = `
        <img src="${e.target.result}" alt="${file.name}">
        <button class="remove-btn" onclick="removePreview(${i})">×</button>
        <div class="caption"><input placeholder="Ibsa..." value="${file.name.replace(/\.[^.]+$/,'')}"></div>`;
      grid.appendChild(card);
    };
    reader.readAsDataURL(file);
  });
  actions.style.display = 'flex';
}

function removePreview(i) { pendingFiles.splice(i,1); handleFiles(pendingFiles); }

function uploadImages() {
  if (!pendingFiles.length) return;

    fd.append('photos[]', file);
    fd.append('titles[]', captions[i] ? captions[i].value : '');
    fd.append('descriptions[]', '');
  });

  bar.style.display = 'block';
  fill.style.width = '10%';

  fetch('upload.php?action=upload', { method: 'POST', body: fd })
    .then(r => r.json())
    .then(data => {
      fill.style.width = '100%';
      setTimeout(() => {
        bar.style.display = 'none';
        fill.style.width = '0%';
        if (data.success) {
          status.innerHTML = `<p style="color:#28a745;margin-top:1rem;">✅ Suuraawwan ${data.uploaded_count} ol fe'ame!</p>`;
          toast(`Suuraawwan ${data.uploaded_count} ol fe'ame!`);
          pendingFiles = [];
          document.getElementById('previewGrid').innerHTML = '';
          document.getElementById('uploadActions').style.display = 'none';
          document.getElementById('fileInput').value = '';
        } else {
          const errs = data.errors ? data.errors.join('<br>') : data.message;
          status.innerHTML = `<p style="color:#e74c3c;margin-top:1rem;">❌ ${errs}</p>`;
          toast(data.message || 'Upload hin milkoofne.', 'error');
        }
      }, 300);
    })
    .catch(err => {
      bar.style.display = 'none';
      status.innerHTML = `<p style="color:#e74c3c;margin-top:1rem;">❌ Server waliin walqunnamuu hin dandeenye. PHP server irratti hojjechaa akka jiru mirkaneessi.</p>`;
      toast('Server waliin walqunnamuu hin dandeenye.', 'error');
    });
}

function changePassword() {
  const oldP = document.getElementById('oldPass').value;
  const newP = document.getElementById('newPass').value;
  const conP = document.getElementById('confirmPass').value;
  const creds = getCreds();
  if (oldP !== creds.pass) { toast("Jecha darbii ammaa dogoggoraa dha.", 'error'); return; }
  if (newP.length < 6)     { toast("Jecha darbii haaraan qubee 6 ol ta'uu qaba.", 'error'); return; }
  if (newP !== conP)        { toast("Jecha darbii haaraan wal hin simne.", 'error'); return; }
  setData(ADMIN_CREDS_KEY, { user: creds.user, pass: newP });
  ['oldPass','newPass','confirmPass'].forEach(id => document.getElementById(id).value = '');
  toast('Jecha darbii jijjiirameera!');
}

}
